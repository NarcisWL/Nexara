How do I implement a custom hook in React?

To implement a custom hook in React, you create a JavaScript function whose name starts with use and that may call other built-in Hooks. Custom hooks allow you to extract component logic into reusable functions.

Here is an example of a common custom hook, useLocalStorage, which syncs state with the browser's local storage:

javascript 

import { useState } from 'react';

export function useLocalStorage(key, initialValue) {
  /* Initialize state with a function to read from local storage once */
  const [storedValue, setStoredValue] = useState(() => {
    try {
      const item = window.localStorage.getItem(key);
      return item ? JSON.parse(item) : initialValue;
    } catch (error) {
      console.error(error);
      return initialValue;
    }
  });

  /* Return a wrapped version of useState's setter */
  const setValue = (value) => {
    try {
      const valueToStore = value instanceof Function ? value(storedValue) : value;
      setStoredValue(valueToStore);
      window.localStorage.setItem(key, JSON.stringify(valueToStore));
    } catch (error) {
      console.error(error);
    }
  };

  return [storedValue, setValue];
}
